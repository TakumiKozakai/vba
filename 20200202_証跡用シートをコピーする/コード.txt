Sub sheetCopy()    Application.ScreenUpdating = False    '選択しているセル数を取得    Dim selectedCaseCnt As Long    selectedCaseCnt = Selection.Count    '選択したセル値を配列に格納    Dim caseNoArray() As Variant    ReDim caseNoArray(selectedCaseCnt)    Dim i As Long    'Cells(行インデックス、列インデックス)    For i = 1 To selectedCaseCnt        caseNoArray(i) = Cells(i + 1, 1).Value    Next    For i = 1 To selectedCaseCnt        'シート選択        Worksheets("エビデンス").Copy After:=Worksheets(Worksheets.Count)        Sheets(Worksheets.Count).Select        Dim sheetName As String     'シート名        Dim caseNo As String        'ケース番号        Dim halfCaseNo As String    'ケース番号（半角）        Dim branchNo As String      'ケース枝番        Dim tmp As Variant          'ケース番号枝番用配列        caseNo = caseNoArray(i)        halfCaseNo = StrConv(caseNo, vbNarrow)        branchNo = InStr(halfCaseNo, "-")        If 0 < branchNo Then            tmp = Split(halfCaseNo, "-")            halfCaseNo = CStr(Val(tmp(0)))            sheetName = Format(halfCaseNo, "000") & "-" & tmp(1)            Range("A1") = "エビデンス" & sheetName            Range("A9") = caseNo        Else            sheetName = Format(halfCaseNo, "000")            Range("A1") = "エビデンス" & sheetName            Range("A9") = caseNo        End If        Sheets(Worksheets.Count).Name = sheetName        Range("A1").Select    NextEnd Sub
