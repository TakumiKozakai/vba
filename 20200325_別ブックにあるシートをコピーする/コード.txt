Sub otherBooksSheetCopy()    'コピー元のブックがあるディレクトリのパスを取得    Const Path As String = "/Users/kozakaitakumi/Workspace/VBA/20200325_別ブックにあるシートをコピーする/"    'ディレクトリ内にある全てのブック名を取得    Dim buf As String    buf = Dir(Path & "*.xlsx")    'コピー先ブックのシート数を取得（末尾にコピーしたいため、シート数を把握）    Dim cnt As Long    cnt = Workbooks("VBA.xlsm").Sheets.Count    'ディレクトリ内にあるコピー対象ブック数分、
    'コピー対象ブックのシートをコピー    Do While buf <> ""        'コピー元ブックを開く        Workbooks.Open Path & buf        'コピー先ブックのシート末尾にコピー元ブックの「○○○」シートをコピー        Workbooks(buf).Worksheets("Eclipse").Copy _            After:=Workbooks("VBA.xlsm").Worksheets(cnt)        'コピーしたシートのシート名をブック名に変更        Workbooks("VBA.xlsm").Worksheets("Eclipse").Select        Workbooks("VBA.xlsm").Worksheets("Eclipse").Name = buf        'コピー元ブックを閉じる        Workbooks(buf).Close        '次のブック名を取得        buf = Dir()        'シート数カウントアップ        cnt = cnt + 1    LoopEnd Sub
